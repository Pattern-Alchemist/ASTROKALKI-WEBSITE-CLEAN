   268тЖТ   268тЖТfunction calculateResult(answers: Record<number, number>): QuizResult {
   269тЖТ   269тЖТ  const scores: Record<string, number> = {};
   270тЖТ   270тЖТ  const matchingByPattern: Record<
   271тЖТ   271тЖТ    string,
   272тЖТ   272тЖТ    { question: QuizQuestion; option: QuizOption }[]
   273тЖТ   273тЖТ  > = {};
   274тЖТ   274тЖТ
   275тЖТ   275тЖТ  for (const q of QUESTIONS) {
   276тЖТ   276тЖТ    const optIdx = answers[q.id];
   277тЖТ   277тЖТ    if (optIdx === undefined) continue;
   278тЖТ   278тЖТ    const opt = q.options[optIdx];
   279тЖТ   279тЖТ    if (!opt) continue;
   280тЖТ   280тЖТ    scores[opt.pattern] = (scores[opt.pattern] ?? 0) + 1;
   281тЖТ   281тЖТ    if (!matchingByPattern[opt.pattern]) matchingByPattern[opt.pattern] = [];
   282тЖТ   282тЖТ    matchingByPattern[opt.pattern].push({ question: q, option: opt });
   283тЖТ   283тЖТ  }
   284тЖТ   284тЖТ
   285тЖТ   285тЖТ  // Sort by score desc, then by atlas order (deterministic tiebreak).
   286тЖТ   286тЖТ  const atlasOrder = new Map(ATLAS_PATTERNS.map((p, i) => [p.slug, i]));
   287тЖТ   287тЖТ  const sorted = Object.entries(scores).sort((a, b) => {
   288тЖТ   288тЖТ    if (b[1] !== a[1]) return b[1] - a[1];
   289тЖТ   289тЖТ    const ai = atlasOrder.get(a[0]) ?? 999;
   290тЖТ   290тЖТ    const bi = atlasOrder.get(b[0]) ?? 999;
   291тЖТ   291тЖТ    return ai - bi;
   292тЖТ   292тЖТ  });
   293тЖТ   293тЖТ
   294тЖТ   294тЖТ  const winnerSlug = sorted[0]?.[0] ?? ATLAS_PATTERNS[0].slug;
   295тЖТ   295тЖТ  const winnerPattern =
   296тЖТ   296тЖТ    ATLAS_PATTERNS.find((p) => p.slug === winnerSlug) ?? ATLAS_PATTERNS[0];
   297тЖТ   297тЖТ
   298тЖТ   298тЖТ  return {
   299тЖТ   299тЖТ    slug: winnerSlug,
   300тЖТ   300тЖТ    pattern: winnerPattern,
   301тЖТ   301тЖТ    scores,
   302тЖТ   302тЖТ    answers,
   303тЖТ   303тЖТ    matchingQuestions: matchingByPattern[winnerSlug] ?? [],
   304тЖТ   304тЖТ  };
   305тЖТ   305тЖТ}
