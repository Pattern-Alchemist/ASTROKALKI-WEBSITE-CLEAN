   158→// ─── Helpers ──────────────────────────────────────────────────────

   160→function formatDate(dateStr: string) {
   161→  try {
   162→    return new Date(dateStr).toLocaleDateString("en-IN", {
   163→      day: "2-digit",
   164→      month: "short",
   165→      year: "numeric",
   166→    });
   167→  } catch {
   168→    return dateStr;
   169→  }
   170→}
   171→
   172→function formatDateTime(dateStr: string) {
   173→  try {
   174→    return new Date(dateStr).toLocaleString("en-IN", {
   175→      day: "2-digit",
   176→      month: "short",
   177→      year: "numeric",
   178→      hour: "2-digit",
   179→      minute: "2-digit",
   180→    });
   181→  } catch {
   182→    return dateStr;
   183→  }
   184→}
   185→
   186→function formatPrice(price: string) {
   187→  const num = parseFloat(price.replace(/[^0-9.]/g, ""));
   188→  if (isNaN(num)) return price;
   189→  return "₹" + num.toLocaleString("en-IN");
   190→}
